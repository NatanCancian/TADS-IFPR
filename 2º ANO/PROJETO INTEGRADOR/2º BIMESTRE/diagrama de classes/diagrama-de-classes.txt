package "Domínio Organizacional" {
    class Fornecedor {
        - Long id
        - String nome
        - String cnpj
        - String email
    }

    class Orgao {
        - Long id
        - String nome
        - String sigla
    }

    class Categoria {
        - Long id
        - String nome
    }
    
    class Documento {
        - Long id
        - String nomeArquivo
        - String urlStorage
        - TipoDocumento tipo
        - String mimeType
        - Long tamanhoBytes
        - String hashIntegridade
        - LocalDateTime dataUpload
    }

    class LogAuditoria {
        - Long id
        - String operacao
        - String entidadeAfetada
        - Long entidadeId
        - String dadosAnteriores
        - String dadosNovos
        - String ip
        - LocalDateTime timestamp
    }
}

package "Núcleo Transacional (Contrato)" {
    class Contrato {
        - Long id
        - String numero
        - String objeto
        - BigDecimal valorTotal
        - LocalDate dataInicio
        - LocalDate dataFim
        - StatusContrato status
        - NivelRisco riscoAtual
    }

    class Aditivo {
        - Long id
        - LocalDate dataAssinatura
        - BigDecimal valorAlterado
        - Integer prazoExtensaoDias
        - TipoAditivo tipo
    }

    class ExecucaoFinanceira {
        - Long id
        - String notaEmpenho
        - String descricaoFatura
        - BigDecimal valorEmpenhado
        - LocalDate dataEmpenho
        - BigDecimal valorLiquidado
        - LocalDate dataLiquidacao
        - BigDecimal valorPago
        - LocalDate dataPagamento
        - StatusExecucao status
    }
}

package "Fiscalização e Inteligência" {
    class Ocorrencia {
        - Long id
        - String descricao
        - Gravidade gravidade
        - LocalDate data
    }

    class Fiscalizacao {
        - Long id
        - LocalDate data
        - String parecer
        - Double notaQualidade
        - Double notaPrazo
        - Double notaConformidade
        - Boolean aprovado
    }
    
    class Alerta {
        - Long id
        - String titulo
        - String mensagem
        - TipoAlerta tipo
        - LocalDateTime dataCriacao
        - Boolean lido
    }

    class Usuario {
        - Long id
        - String nome
        - String matricula
        - Perfil perfil
        - String senhaHash
    }
}

package "Camada de Serviços (Backend Inteligente)" {
    
    class AutenticacaoService {
        + autenticar(login, senha): String
        + validarToken(token): Boolean
    }

    class CalculadoraRiscoService {
        + calcularScore(Contrato c): NivelRisco
    }
    
    class FinanceiroService {
        + calcularSaldoRestante(Contrato c): BigDecimal
        + calcularPendentePagamento(Contrato c): BigDecimal
        + processarLiquidacao(ExecucaoFinanceira ef, BigDecimal valor): void
    }
    
    class ContratoMetricaService {
        + gerarMetricasDashboard(Contrato c): DTO
    }
    
    class MonitorVigenciaJob {
        + executarVerificacaoDiaria(): void
    }
    
    class SincronizadorEloweb {
        + executarSincronizacaoIncremental(): void
    }
    
    class ExportadorPNCP {
        + gerarLogPadraoPNCP(Contrato c): String
    }
}

' Enumerações formatadas com quebra de linha
enum StatusContrato { 
    ATIVO
    ENCERRADO
    INATIVO
    SUSPENSO 
}

enum NivelRisco { 
    BAIXO
    MEDIO
    ALTO 
}

enum TipoAlerta { 
    RISCO_ELEVADO
    SALDO_CRITICO
    VENCIMENTO_PROXIMO 
}

enum Perfil { 
    GESTOR
    FISCAL
    AUDITOR
    ADMIN 
}

enum Gravidade { 
    BAIXA
    MEDIA
    ALTA 
}

enum TipoAditivo { 
    VALOR
    PRAZO
    ESCOPO 
}

enum TipoDocumento { 
    ATA
    EMPENHO
    ADITIVO
    EXTRATO 
}

enum StatusExecucao { 
    EMPENHADO
    LIQUIDADO
    PAGO
    CANCELADO 
}

' Relacionamentos Organizacionais
Contrato "1" --> "1" Fornecedor : contratada
Contrato "1" --> "1" Orgao : contratante
Contrato "1" --> "1" Categoria : classificado em

' Relacionamentos de Estrutura
Contrato "1" *-- "0..*" Aditivo : possui
Contrato "1" *-- "0..*" ExecucaoFinanceira : registra gasto
Contrato "1" *-- "0..*" Documento : possui anexos

' Relacionamentos de Inteligência e Fiscalização
Contrato "1" *-- "0..*" Ocorrencia : gera histórico
Contrato "1" *-- "0..*" Fiscalizacao : é vistoriado
Contrato "1" *-- "0..*" Alerta : gera

' Relacionamentos de Usuário e Auditoria
Usuario "1" -- "0..*" Fiscalizacao : realiza
Usuario "1" -- "0..*" Alerta : visualiza
Usuario "1" -- "0..*" LogAuditoria : executa

' Dependências de Serviços
AutenticacaoService ..> Usuario : valida
MonitorVigenciaJob ..> Contrato : verifica datas
CalculadoraRiscoService ..> Contrato : analisa
FinanceiroService ..> Contrato : processa finanças
ContratoMetricaService ..> Contrato : consolida dados
SincronizadorEloweb ..> Contrato : atualiza incrementalmente
ExportadorPNCP ..> Contrato : exporta
